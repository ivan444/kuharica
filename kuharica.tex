\begin{recipe}{Quiche sa špinatom i feta sirom}%
  {4 porcije}%
  {\showclock{0}{50} (10$m$ pripreme, 5$m$ \fryingpan, 35$m$ \oven) \textleaf}
\ingredient[1]{}{srednji luk}
\ingredient[]{}{ulje}
  Sitno nareži luk i poprži na ulju (lagana vatra, 5$m$). Ostavi sa strane da
  se malo ohladi.

\ingredient[4]{}{jaja}
\ingredient[250]{ml}{vrhnja za kuhanje}
\ingredient[200]{g}{\grater feta sira}
\ingredient[\fr18]{\Spoon}{muškatnog orašćića}
\ingredient[\fr18]{\Spoon}{papra}
  Izmuti jaja, dodaj vrhnje za kuhanje. Naribaj feta sir i dodaj u smjesu.
  Dodaj muškatni orašćić i papar.

\ingredient[200]{g}{smrznutog špinata}
  Otopi špinat u mikrovalnoj. Iscjedi vodu iz špinata kroz cjedilo za čaj.
  Dodaj u smjesu.

\newstep
  Dodaj poprženi luk u smjesu. Sve dobro pomješaj.

\ingredient[1]{}{tijesto za quiche\\
  \O\ 33$cm$}
  Stavi tijesto u okruglu posudu. Nalij smjesu.

\ingredient[]{}{\oven}
  Zagrij pećnicu na 180\0 C. Peci 35$m$.

\end{recipe}

\begin{recipe}{Rižoto s piletinom i povrćem}%
  {4 porcije}%
  {\showclock{0}{35} (20$m$ pripreme, 15$m$ \pot)}
\ingredient[150]{g}{slanine}
\ingredient[]{}{maslinovo ulje}
  Nareži slaninu na male komadiće i poprži na ulju.

\ingredient[200]{g}{pilećih prsa}
  Sitno nareži piletinu i dodaj na tavu.

\ingredient[1]{}{srednji luk}
\ingredient[1]{}{mrkva}
\ingredient[1]{}{tikvica}
  Sitno nareži luk i dodaj na tavu (lagana vatra, 2-3$m$). Sitno nareži povrće
  (posebno sitno mrkvu) i dodaj na tavu.

\ingredient[2]{}{češnjaka}
\ingredient[1]{\Spoon}{timjan}
\ingredient[400]{g}{Arborio riže}
  Dodaj protisnuti češnjak, timjan i i rižu na tavu. Miješaj 2-3$m$.

\ingredient[100]{ml}{bijelog vina}
\ingredient[1]{l}{vode}
\ingredient[2]{}{pileće kocke}
  Dodaj vino i kuhaj dok ne ispari. Dodaj vodu. Umješaj pileće kocke. Dobro
  promješaj.

\ingredient[]{}{\pot}
  Sve stavi u ekspres lonac i kuhaj 7$m$ od trenutka kad dostigne 40$kPa$.

\ingredient[25]{g}{maslaca}
\ingredient[4]{\Spoon}{parmezana}
\ingredient[\fr14]{\Spoon}{papra}
  Dodaj maslac, parmezan i papar. Dobro promješaj.

  Posluži sa dodatnim parmezanom.

\freeform
  \ding{43} Piletina, tikvice i mrkva se može zamjeniti sa 200$g$ šampinjona
  ili 500$g$ šparoga.

\end{recipe}

\begin{recipe}{Palačinke od cikle}%
  {10 palačinki}%
  {\showclock{1}{30} (15$m$ pripreme, 1$h$ \pot, 15$m$ \fryingpan ) \textleaf}
\ingredient[1]{}{mala cikla}
\ingredient[]{}{voda}
  Kuhaj cijelu ciklu (neoguljenu) u vodi 1$h$. Oguli je nakon što se cikla
  skuha i ohladi. (moguće je skuhati veću količinu i zalediti)

  Nareži ciklu na manje komade.

\ingredient[2]{\Spoon}{maslaca}
  Otopi maslac.

\ingredient[250]{ml}{mlijeka}
\ingredient[120]{g}{brašna}
\ingredient[2]{}{jaja}
\ingredient[1]{\Spoon}{vanilije}
\ingredient[1]{\Spoon}{šećera}
  Smiksaj ciklu, maslac i ostale sastojke u blenderu.

\ingredient[]{}{\fryingpan}
  Poprži na tavi, na malo ulja.

\end{recipe}

\begin{recipe}{Pomfrie of batata}%
  {3 porcije}%
  {\showclock{1}{20} (20$m$ pripreme, 30$m$ namakanja, 30$m$ \oven ) \textleaf}
\ingredient[1]{}{veliki batat}
\ingredient[]{}{voda}
  Nareži batat na prutiće debljine 1$cm$.

  Natopi ih u vodi bar 30$m$.

\ingredient[2]{\Spoon}{maslinovog ulja}
  Osuši batat ubrusima i naulji maslinovim uljem.

\ingredient[2]{\Spoon}{škroba}
\ingredient[\fr12]{\Spoon}{češnjaka u prahu}
\ingredient[\fr12]{\Spoon}{paprike u prahu}
\ingredient[\fr12]{\Spoon}{papra}
\ingredient[\fr14]{\Spoon}{soli}
  Pomješaj sve suhe sastojke. Uvaljaj nauljeni batat u njih.

\ingredient[]{}{\oven}
  Zagrij pećnicu na 215\0 C.

  Poslaži krumpire na papir za pečenje. Peci 15$m$, preokreni pa peci još
  10-15$m$.

\freeform
  \ding{43} Mrkva je dobra zamjena za batat.

\end{recipe}

\begin{recipe}{Pulled pork}%
  {4 porcije}%
  {\showclock{4}{25} (10$m$ pripreme, 4$h$ \oven, 15$m$ \fryingpan )}
\ingredient[500]{g}{pork shoulder}
\ingredient[1]{\Spoon}{paprike u prahu}
\ingredient[1]{\Spoon}{soli}
\ingredient[1]{\Spoon}{češnjaka u prahu}
  Namaži začine po mesu.

\ingredient[1]{}{luk}
  Sitno nareži luk.

\ingredient[60]{ml}{vode}
\ingredient[]{}{\pot}
  Stavi meso, luk i vodu u slow cooker. Kuhaj 3-4$h$ na high.

\newstep
  Izvadi meso i razdvoji sa 2 vilice. Baci vodu.

\ingredient[230]{ml}{BBQ sauce}
\ingredient[200]{ml}{passata}
  Dodaj BBQ sauce i passatu mesu te kuhaj 10-15$m$.

  Posluži uz burger bunse i coleslaw.
\end{recipe}

\begin{recipe}{Lazanje}%
  {6 porcija}%
  {\showclock{1}{25} (15$m$ pripreme, 45$m$ \fryingpan, 25$m$ \oven )}
\ingredient[1]{}{luk}
\ingredient[1]{}{mrkva}
\ingredient[1]{}{češnjak}
\ingredient[]{}{maslinovo ulje}
\ingredient[]{}{\fryingpan}
  {\bf Bolonjez umak}\newline
  Sitno nareži luk, mrkvu i češnjak. Zagrij ulje na tavi te dinstaj luk dok ne
  postane proziran. Dodaj mrkvu i češnjak te dinstaj par minuta.

\Ing{400-500g\qquad mljevene govedine}
  Dodaj mljeveno meso u tavu. Razbij velike komade vilicom. Dinstaj dok nema
  više crvenih dijelova.

\ingredient[400]{ml}{passate}
\ingredient[]{}{sol}
\ingredient[2]{lista}{lovora}
\ingredient[]{}{peršin}
\ingredient[]{}{timjan}
\ingredient[]{}{papar}
  Dodaj passatu i začine. Lagano kuhaj 30$m$.

\ingredient[2]{žlice}{maslaca}
\ingredient[3]{žlice}{ulja}
\ingredient[2-3]{žlice}{brašna}
\ingredient[]{}{\pot}
  {\bf Bešamel umak}\newline
  U novom loncu napravi zapršku od ulja, maslaca i brašna.

\Ing{600-700ml\qquad mlijeka}
\ingredient[]{}{sol}
\ingredient[]{}{papar}
\ingredient[]{}{muškatni oraščić}
\ingredient[]{}{vegeta}
  Dodaj u zapršku mlijeko i začine. Kuhaj na srednjoj vatri i miješati dok se
  ne zgusne.

\ingredient[]{}{listovi lazanja}
\ingredient[100]{g}{Mozzarella sira}
\ingredient[]{}{\pan}
  {\bf Slaganje}\newline
  Stavi malo (1-2 žlice) bolonjez umaka na dno posude. Zatim, dva puta: red
  listova, bolonjez umak, bešamel umak.

  Na vrh stavi red listova, bešamel umaka, te sir.

\ingredient[]{}{\oven}
  Zagrij pećnicu na 180\0 C. Peci 20-25$m$.

\end{recipe}

\begin{recipe}{Uskršnja pogača / pinca}%
  {2 srednje pogače}%
  {\showclock{4}{25} (4.5$h$ pripreme, 30$m$ \oven )}
\ingredient[500]{g}{brašna}
\ingredient[50]{g}{svježeg kvasca / vrećica suhog)}
\ingredient[60]{g}{šećera}
\ingredient[1]{dcl}{toplog mlijeka}
  Prosijaj brašno i napravi udubinu. Unutra uspi šećer, kvasac i mlijeko.
  Lagano pomiješaj. Pokrij i pusti da se diže 30$m$.

\ingredient[3]{}{jaja (na sobnoj temperaturi)}
\ingredient[60]{g}{šećera}
\ingredient[7]{g}{soli}
\ingredient[1]{}{kora naranče}
\ingredient[1]{}{kora limuna}
  U smjesu dodati jaja, šećer, sol i arome (potencijalni dodatci: rum, Maraschino, arome za sirnicu, arome ruže). Miješati mikserom dok tijesto ne postane homogeno i glatko, te se odvaja od posude. Pokrij i pusti da se diže dok se ne udvostruči (45-60$m$).

\ingredient[100]{g}{maslaca}
  Omekšaj maslac i pomiješaj sa tijestom. Miksaj dok tijesto ponovo ne postane
  homogeno i glatko. Pokrij i pusti da se diže dok se ne udvostruči (45-60$m$).

\newstep
  Razdvoji tijesto na dva dijela i oblikuj kugle. Pokrij i pusti da se diže dok
  se ne udvostruči (45-60$m$).

\ingredient[1]{}{jaje}
\ingredient[10]{g}{šećera}
  Izmuti jaje. Šećer malo navlaži i promiješaj da se stvore grudvice. Na vrhu
  tijesta zareži (od sredine prema van) tri kratke linije. Premaži jajima i
  pospi šećerom.

\ingredient[]{}{\oven}
  Zagrij pećnicu na 160\0 C. Peci 30$m$.
\end{recipe}
